   514→function parseForm(code,count){
   515→  // code '4-3-3' -> filas atrás→adelante. válida si suma+arquero == count
   516→  const rows=code.split('-').map(Number);
   517→  if(rows.reduce((a,b)=>a+b,0)+1!==count) return null;
   518→  return rows;
   519→}
   520→function formationPositions(code,count,isBlue){
   521→  let rows=parseForm(code,count);
   522→  if(!rows){ // genérica
   523→    const n=count-1, per=Math.ceil(n/3);
   524→    rows=[];let left=n;
   525→    while(left>0){rows.push(Math.min(per,left));left-=per}
   526→  }
   527→  const pos=[[0.5,0.93]]; // arquero
   528→  // todos en campo propio: la línea más adelantada queda detrás de la mitad
   529→  const yBack=0.82, yFront=S.game===5?0.62:0.57;
   530→  rows.forEach((cnt,r)=>{
   531→    const y= rows.length===1? (yBack+yFront)/2 : yBack-(yBack-yFront)*r/(rows.length-1);
   532→    for(let j=0;j<cnt;j++){
   533→      const x=(j+1)/(cnt+1);
   534→      pos.push([0.5+(x-0.5)*0.92, y]);
   535→    }
   536→  });
   537→  return isBlue?pos:pos.map(([x,y])=>[1-x,1-y]);
   538→}

   883→function posAt(kf,t){
   884→  // Interpolación Catmull-Rom: pasa suave por todos los puntos grabados,
   885→  // sin el efecto "frena-arranca" de animar tramo por tramo.
   886→  const n=kf.length;
   887→  if(n===1||t<=kf[0].t)return kf[0];
   888→  if(t>=kf[n-1].t)return kf[n-1];
   889→  let i=1;while(kf[i].t<t)i++;
   890→  const p1=kf[i-1],p2=kf[i];
   891→  const p0=kf[i-2]||p1,p3=kf[i+1]||p2;
   892→  const u=(t-p1.t)/Math.max(1,p2.t-p1.t),u2=u*u,u3=u2*u;
   893→  const cr=(a,b,c,d)=>0.5*((2*b)+(-a+c)*u+(2*a-5*b+4*c-d)*u2+(-a+3*b-3*c+d)*u3);
   894→  return{x:clamp(cr(p0.x,p1.x,p2.x,p3.x),0.02,0.98),
   895→         y:clamp(cr(p0.y,p1.y,p2.y,p3.y),0.02,0.98)};
   896→}

   949→function completePlay(baseTracks,duration,style,attOverride){
   950→  const SAMPLE=220;
   951→  const ids=Object.keys(baseTracks);
   952→  const start={},active={};
   953→  ids.forEach(id=>{start[id]=baseTracks[id][0];active[id]=pathLen(baseTracks[id])>0.06});
   954→
   955→  const ball=baseTracks['ball'];
   956→  const ballMoved=ball&&pathLen(ball)>0.06;
   957→
   958→  // referencia de juego: la pelota, o si no se movió, el jugador más activo
   959→  let refTrack=ball&&ball.length?ball:[{t:0,x:.5,y:.5}];
   960→  if(!ballMoved){
   961→    let bl=0;
   962→    S.players.forEach(p=>{
   963→      if(p.ball)return;
   964→      const l=pathLen(baseTracks[p.id]||[]);
   965→      if(l>bl){bl=l;refTrack=baseTracks[p.id]}
   966→    });
   967→  }
   968→  const ballAt=t=>posAt(refTrack,t);
   969→
   970→  // ¿quién ataca? 1) lo que elija el usuario, 2) hacia dónde va la pelota,
   971→  // 3) qué equipo avanzó más, 4) quién tiene la pelota más cerca
   972→  let att=(attOverride&&attOverride!=='auto')?attOverride:null;
   973→  if(!att&&ballMoved){
   974→    const dy=ball[ball.length-1].y-ball[0].y;
   975→    if(Math.abs(dy)>0.05)att=dy<0?'blue':'red'; // sube → ataca azul
   976→  }
   977→  if(!att){
   978→    const adv={blue:0,red:0};
   979→    S.players.forEach(p=>{
   980→      if(p.ball)return;const kf=baseTracks[p.id];if(!kf)return;
   981→      const d=(kf[0].y-kf[kf.length-1].y)*(p.team==='blue'?1:-1);
   982→      adv[p.team]+=Math.max(0,d);
   983→    });
   984→    if(Math.abs(adv.blue-adv.red)>0.02)att=adv.blue>adv.red?'blue':'red';
   985→  }
   986→  if(!att){
   987→    const b0=ballAt(0);let bd=9;
   988→    S.players.forEach(p=>{
   989→      if(p.ball||p.gk)return;
   990→      const d=Math.hypot(start[p.id].x-b0.x,start[p.id].y-b0.y);
   991→      if(d<bd){bd=d;att=p.team}
   992→    });
   993→  }
   994→  att=att||'blue';
   995→  const def=att==='blue'?'red':'blue';
   996→  const dAtt=dirOf(att);
   997→
   998→  // presor: defensor más cercano a la pelota al inicio (no arquero)
   999→  let presser=null,pd=9;
  1000→  S.players.forEach(p=>{
  1001→    if(p.team!==def||p.gk||p.ball)return;
  1002→    const b0=ballAt(0),d=Math.hypot(start[p.id].x-b0.x,start[p.id].y-b0.y);
  1003→    if(d<pd){pd=d;presser=p.id}
  1004→  });
  1005→
  1006→  // corredores especiales del equipo atacante (estilo)
  1007→  let winger=null,striker=null;
  1008→  if(style==='attack'||style==='wings'||style==='balanced'){
  1009→    const b0=ballAt(0),side=b0.x<0.5?'L':'R';
  1010→    let wBest=0;
  1011→    S.players.forEach(p=>{
  1012→      if(p.team!==att||p.gk||active[p.id])return;
  1013→      const sx=start[p.id].x;
  1014→      const sameSide=side==='L'?sx<0.5:sx>=0.5;
  1015→      const wide=Math.abs(sx-0.5);
  1016→      if(sameSide&&wide>wBest){wBest=wide;winger=p.id}
  1017→    });
  1018→    let sBest=9;
  1019→    S.players.forEach(p=>{
  1020→      if(p.team!==att||p.gk||active[p.id]||p.id===winger)return;
  1021→      const gy=att==='blue'?0:1;
  1022→      const d=Math.abs(start[p.id].y-gy);
  1023→      if(d<sBest){sBest=d;striker=p.id}
  1024→    });
  1025→  }
  1026→
  1027→  const depth= style==='attack'?0.26 : style==='defense'?0.07 : style==='wings'?0.16 : 0.16;
  1028→  const out={};
  1029→  const steps=Math.max(2,Math.ceil(duration/SAMPLE));
  1030→
  1031→  for(const p of S.players){
  1032→    const id=p.id;
  1033→    if(active[id]||p.ball&&ballMoved){out[id]=baseTracks[id];continue}
  1034→    if(p.ball){ // pelota quieta: queda donde está
  1035→      out[id]=baseTracks[id]||[{t:0,x:p.x,y:p.y},{t:duration,x:p.x,y:p.y}];continue;
  1036→    }
  1037→    const a=start[id],kf=[];
  1038→    for(let s=0;s<=steps;s++){
  1039→      const t=Math.min(duration,s*SAMPLE);
  1040→      const b=ballAt(t);
  1041→      const prog=clamp(((ballAt(0).y-b.y)*dAtt*-1)/0.55,0,1); // avance de la pelota 0→1
  1042→      let x,y;
  1043→
  1044→      if(p.gk){ // arquero: se desplaza lateralmente siguiendo la pelota, pegado a su arco
  1045→        x=0.5+(b.x-0.5)*0.28;
  1046→        y=p.team==='blue'?clamp(a.y,0.87,0.94):clamp(a.y,0.06,0.13);
  1047→      }
  1048→      else if(p.team===att){
  1049→        // compañero de apoyo
  1050→        x=a.x+(b.x-a.x)*0.22;
  1051→        y=a.y + dAtt*depth*prog;
  1052→        if(id===winger){ // desborde por banda
  1053→          const tx=ballAt(0).x<0.5?0.10:0.90;
  1054→          x=a.x+(tx-a.x)*Math.min(1,prog*1.5);
  1055→          y=a.y + dAtt*(depth+0.16)*prog;
  1056→        }
  1057→        if(id===striker){ // ataque al área / primer palo
  1058→          const gy=att==='blue'?0.12:0.88;
  1059→          x=a.x+(0.5+(b.x-0.5)*0.4-a.x)*prog;
  1060→          y=a.y+(gy-a.y)*Math.min(1,prog*1.25);
  1061→        }
  1062→        if(style==='wings'&&id!==winger&&Math.abs(a.x-0.5)>0.18){
  1063→          x=a.x+( (a.x<0.5?0.13:0.87) -a.x)*0.5*prog;
  1064→        }
  1065→        // no pisar la pelota
  1066→        const dball=Math.hypot(x-b.x,y-b.y);
  1067→        if(dball<0.085){const f=(0.085-dball);x+= (x-b.x)/Math.max(0.01,dball)*f;y+=(y-b.y)/Math.max(0.01,dball)*f}
  1068→      }
  1069→      else{ // defensor: el bloque bascula hacia la pelota y se repliega; el más cercano presiona
  1070→        const retreat=def==='blue'?0.10:-0.10; // hacia su propio arco
  1071→        x=a.x+(b.x-a.x)*0.30;
  1072→        y=a.y+retreat*prog;
  1073→        if(style==='defense'){
  1074→          x=a.x+(b.x-a.x)*0.18+(0.5-a.x)*0.18;
  1075→          y=a.y+retreat*1.4*prog;
  1076→        }
  1077→        if(id===presser){ // sale a presionar al portador, tapando el camino al arco
  1078→          const off=def==='blue'?0.055:-0.055;
  1079→          const reach=Math.min(1,0.25+prog*1.1);
  1080→          x=a.x+(b.x-a.x)*reach;
  1081→          y=a.y+(b.y+off-a.y)*reach;
  1082→        }
  1083→      }
  1084→      kf.push({t,x:clamp(x,0.03,0.97),y:clamp(y,0.03,0.97)});
  1085→    }
  1086→    out[id]=kf;
  1087→  }
  1088→  return {tracks:out, att};
  1089→}
